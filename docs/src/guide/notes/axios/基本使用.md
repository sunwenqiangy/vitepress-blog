# 基本使用

`axios()`

```javascript
axios({
  // 请求类型
  method: "",
  // url
  url: "",
  // 请求体
  data: {}
})
```



## 发送get请求

```javascript
// 查询数据
const getApi = () => {
  axios({
    // 请求类型
    method: "GET",
    // URL
    url: "http://127.0.0.1:3000/posts",
  }).then((response) => {
    console.log(response);
  });
};

// GET请求简写方式
const getApi = () => {
  axios("http://127.0.0.1:3000/posts").then((response) => {
    console.log(response);
  });
};
```



## 发送post请求

```javascript
// 添加数据
const postApi = () => {
  const params = {
    author: "dancy",
    title: "学习axios",
  };
  axios({
    // 请求类型
    method: "POST",
    // URL
    url: "http://127.0.0.1:3000/posts",
    // 设置请求体
    data: params,
  }).then((response) => {
    console.log(response);
  });
};
```



## 发送put请求

```javascript
// 更新第二条数据
const putApi = () => {
  axios({
    // 请求类型
    method: "PUT",
    // URL
    url: "http://127.0.0.1:3000/posts/2",
    // 设置请求体
    data: {
      author: "dancy",
      title: "学习前端",
    },
  }).then((response) => {
    console.log(response);
  });
};
```



## 发送delete请求

```javascript
// 删除第二条数据
const deleteApi = () => {
  axios({
    // 请求类型
    method: "DELETE",
    // URL
    url: "http://127.0.0.1:3000/posts/2",
  }).then((response) => {
    console.log(response);
  });
};
```
